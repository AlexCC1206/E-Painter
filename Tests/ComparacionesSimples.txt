// Prueba simplificada de operaciones booleanas
Spawn(10, 10)
Color("Black")
Size(2)

// Variables para pruebas
a <- 5
b <- 3
c <- 5

// Para usar en visualización
lineaTrue <- 5   // Longitud de línea si la condición es true
lineaFalse <- 2  // Longitud de línea si la condición es false

// Comparaciones de igualdad
esIgual <- a == c        // true
noEsIgual <- a == b      // false

// Dibujamos líneas horizontales de diferentes longitudes según el resultado
// Línea para esIgual (debe ser longitud 5)
DrawLine(1, 0, esIgual * lineaTrue + (1 - esIgual) * lineaFalse)

// Línea para noEsIgual (debe ser longitud 2)
DrawLine(0, 1, noEsIgual * lineaTrue + (1 - noEsIgual) * lineaFalse)

// Comparaciones de mayor/menor
esMayor <- a > b         // true
esMenor <- a < b         // false

// Dibujamos líneas verticales
// Línea para esMayor (debe ser longitud 5)
DrawLine(0, 1, esMayor * lineaTrue + (1 - esMayor) * lineaFalse)

// Línea para esMenor (debe ser longitud 2)
DrawLine(1, 0, esMenor * lineaTrue + (1 - esMenor) * lineaFalse)
