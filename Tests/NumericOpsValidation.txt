// Prueba de operaciones numéricas post-análisis
Spawn(10, 10)
Color("Black")
Size(2)

// Asignaciones simples
a <- 4
b <- 2

// Operaciones aritméticas básicas
suma <- a + b       // 6
resta <- a - b      // 2
multi <- a * b      // 8
div <- a / b        // 2
modulo <- a % b     // 0

// Usar resultados en comandos
DrawLine(1, 0, suma)     // Dibuja línea horizontal de 6 unidades
DrawLine(0, 1, resta)    // Dibuja línea vertical de 2 unidades
Size(resta)              // Cambia tamaño a 2
DrawCircle(0, 0, 1)      // Dibuja un círculo con radio 1

// Operaciones más complejas
c <- (a + b) * (a - b)   // (4+2)*(4-2) = 6*2 = 12
d <- 8 / (2 + 2)         // 8/4 = 2
DrawCircle(div, d, 1)    // Dibuja círculo en posición (2,2) con radio 1
