// Prueba de operaciones matemáticas
Spawn(10, 10)
Color("Black")
Size(2)

// Operaciones básicas
a <- 5
b <- 3
c <- a + b  // debe ser 8
d <- a - b  // debe ser 2
e <- a * b  // debe ser 15
f <- a / b  // debe ser 1.67
g <- a % b  // debe ser 2

// Usar los valores en dibujos
DrawLine(1, 0, c)  // línea de 8 unidades
DrawLine(0, 1, d)  // línea de 2 unidades
DrawCircle(0, 0, g)  // círculo de radio 2
