// Prueba completa de comandos de dibujo
Spawn(10, 10)
Color("Black")
Size(2)

// DrawLine - Líneas en diferentes direcciones
DrawLine(5, 0, 3)     // Horizontal derecha
DrawLine(-5, 0, 3)    // Horizontal izquierda
DrawLine(0, 5, 3)     // Vertical arriba
DrawLine(0, -5, 3)    // Vertical abajo
DrawLine(3, 3, 3)     // Diagonal derecha-arriba
DrawLine(-3, -3, 3)   // Diagonal izquierda-abajo

// DrawCircle - Círculos con diferentes radios
Color("Blue")
DrawCircle(0, 0, 2)   // Círculo centrado en origen
DrawCircle(5, 5, 1)   // Círculo pequeño
DrawCircle(-5, -5, 3) // Círculo grande

// DrawRectangle - Rectángulos con diferentes dimensiones
Color("Red")
DrawRectangle(0, 0, 4, 2, 1)    // Rectángulo horizontal
DrawRectangle(5, 5, 2, 4, 1)    // Rectángulo vertical
DrawRectangle(-5, -5, 3, 3, 1)  // Cuadrado

// Fill - Rellenar áreas
Color("Green")
Fill(1, 1)   // Rellenar un área

// Cambios de tamaño de pincel
Size(1)
Color("Purple")
DrawCircle(2, 2, 1)

Size(3)
DrawCircle(4, 4, 1)

Size(5)
DrawCircle(6, 6, 1)

// Secuencia de figuras para dibujar un patrón
Size(2)
Color("Yellow")
DrawLine(0, 0, 8)       // Trazado en zigzag
DrawLine(2, 2, 3)
DrawLine(-2, 2, 3)
DrawLine(2, -2, 3)

// Crear una figura compleja
Color("Orange")
DrawRectangle(0, 0, 10, 10, 1)  // Marco exterior
DrawCircle(5, 5, 3)            // Círculo interior
