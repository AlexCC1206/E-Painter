// Prueba de formas geométricas
Spawn(10, 10)
Color("Black")
Size(1)

// Dibujar un cuadrado
DrawLine(1, 0, 5)
DrawLine(0, 1, 5)
DrawLine(-1, 0, 5)
DrawLine(0, -1, 5)

// Dibujar un círculo
Spawn(3, 3)
Color("Blue")
DrawCircle(0, 0, 2)

// Usar variables
x <- 10 + 5
y <- x - 3
Spawn(x, y)
Color("Black")
DrawLine(1, 1, 3)
