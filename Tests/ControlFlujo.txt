// Prueba de saltos condicionales y etiquetas
Spawn(10, 10)
Color("Black")
Size(2)

// Variables para controlar el flujo
a <- 5
b <- 3
contador <- 0

// Etiqueta principal de inicio
inicio
contador <- contador + 1

// Probar diferentes condiciones
GoTo [mayorQue] (a > b)
GoTo [menorQue] (a < b)
GoTo [iguales] (a == b)
GoTo [fin]  // Si ninguna condición se cumple

mayorQue
// a es mayor que b (condición verdadera)
Color("Green")
DrawLine(contador, 0, 2)
GoTo [inicio] (contador < 3)
GoTo [fin]

menorQue
// a es menor que b (condición falsa, no debería ejecutarse)
Color("Red")
DrawLine(0, contador, 2)
GoTo [fin]

iguales
// a es igual a b (condición falsa, no debería ejecutarse)
Color("Blue")
DrawCircle(contador, contador, 1)
GoTo [fin]

fin
// Finalizar con confirmación visual
Color("Black")
Size(3)
DrawCircle(0, 0, contador)  // El círculo debería tener radio 3 si todo funcionó correctamente
